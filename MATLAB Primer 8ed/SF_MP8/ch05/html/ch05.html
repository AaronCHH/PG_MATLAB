5.9 The bsxfun and repmat functions

A = rand(5)
C = A - mean(A(:))

n = size(A,1)
C = A - repmat(mean(A), n, 1)

C = bsxfun(@minus, A, mean(A))
mean(C)

C = bsxfun(@minus, A, mean(A,2))
mean(C,2)

%{
  @plus + @minus
  @times . * @power .^
  @rdivide ./ @ldivide .\
  @lt < @gt >
  @le <= @ge >=
  @eq == @ne ~=
  @max maximum @min minimum
%}

C = bsxfun(@minus, A, min(A))
min(C)

d = max(abs(A), [], 2)
C = bsxfun(@ldivide, d, A)
C = diag(d) \ A
n = size(A,1)
C = spdiags(d,0,n,n) \ A
