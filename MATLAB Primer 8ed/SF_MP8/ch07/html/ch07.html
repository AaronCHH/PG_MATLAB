7.4 The switch statement

switch x
  case 0
    disp ('x is zero') ;
  case {2, 4, 6, 8}
    disp ('x is 2,4,6, or 8') ;
    x = x / 2 ;
  otherwise
    disp ('x is something else') ;
    x = 0 ;
end

7.6 Matrix expressions (if and while)

A = [ 1 2 ; 3 4 ]
B = [ 2 3 ; 3 5 ]

if A == B
  disp('A and B are equal')
end

if any(any(A ~= B))
  disp('A and B are not equal')
end

if A == B else
  disp('A and B are not equal')
end

if A ~= B
  disp('not what you think')
end

isequal('a', 'bb')
all('a' == 'bb')
