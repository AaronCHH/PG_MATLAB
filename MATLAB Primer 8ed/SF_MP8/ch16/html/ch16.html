t = .01:.01:20*pi ;
x = cos(t) ;
y = sin(t) ;
z = t.^3 ;
plot3(x, y, z)

16.2 Mesh and surface plots

xx = -2:.2:2 ;
yy = xx ;
[x, y] = meshgrid(xx, yy) ;
z = exp(-x.^2 - y.^2) ;
mesh(z)

16.3 Parametrically defined surfaces

figure(1) ; clf
t = linspace(0, 2*pi, 512) ;
[u,v] = meshgrid(t) ;

a = -0.2 ; b = .5 ; c = .1 ;
n = 2 ;
x = (a*(1-v/(2*pi)) .* (1+cos(u)) + c) .* cos(n*v) ;
y = (a*(1-v/(2*pi)) .* (1+cos(u)) + c) .* sin(n*v) ;
z = b*v/(2*pi) + a*(1-v/(2*pi)) .* sin(u) ;

surf(x,y,z,y-2*x)
shading interp
% shading faceted
% shading flat

[x,y,z,v] = flow ;

slice(x, y, z, v, [3 8], 0, 0)
